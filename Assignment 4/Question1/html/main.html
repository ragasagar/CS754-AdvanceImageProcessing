Setting x value

x = zeros(n,1);
random_index = randi([0,n], 18,1);
for i = 1: size(random_index,1)
    x(random_index(i)) =  randi([0,1000]);
end

A = double(rand(m, n)<.5);
for i = 1:m
    for j = 1:n
        if A(i,j) == 1
            A(i,j) = 1/sqrt(m);
        else
            A(i,j) = -1/sqrt(m);
        end
    end
end
R = A(1:r_m,:);
V = A(r_m+1:m,:);

sigma = 0.05 * sum(norm(A*x)/m);
x_noise = x+ sigma*randn(size(x));

y=A*x_noise;
lambdas = [0.0001 0.0005  0.001 0.005  0.01 0.05 0.1 0.5 1 2 5 10 15 20 30 50 100];
errors = zeros(size(lambdas,1));
rmses = zeros(size(lambdas,1));
for i = 1:size(lambdas,2)
    [x_pred, status] = l1_ls(R,y(1:r_m),lambdas(i), 0.01);

    %validation error check
    y_pred = V*x_pred;
    error = sum((y(r_m+1:200) - y_pred).^2)/norm(V);
    errors(i) = error;

    %rmse error check
    rmses(i) = norm(x_pred-x,2)/norm(x,2);
end
figure;
plot(log(lambdas), errors );

title('VE vs logarithm lambdas');
ylabel('VE');
xlabel('log(lambda)');

figure;
plot(log(lambdas), rmses );

title('RMSE vs logarithm lambdas');
